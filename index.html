# 快速开始

# 1.安装组件库

npm i imooc-ui

# 2.引入组件库

// 全部引入
import 'imooc-ui/dist/css/index.css'
import MUI from "imooc-ui";
Vue.use(MUI);

// 按需引入
import 'imooc-ui/dist/css/demo.css'
import { Demo } from "imooc-ui";
Vue.use(Demo);
